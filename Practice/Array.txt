void inputArray(int arr[], int size)
{
    for (int i = 0; i < size; i++)
    {
        cin >> arr[i];
    }
    return;
}
---------------------------------------------------------------------------
2. Output the Array
---------------------------------------------------------------------------
void printArray(int arr[], int size)
{
    for (int i = 0; i < size; i++)
    {
        cout << arr[i] << " ";
    }
    return;
}
---------------------------------------------------------------------------
3. Largest Element in Array
---------------------------------------------------------------------------
int largestElement(int arr[], int size)
{
    int largest = arr[0];
    for (int i = 1; i < size; i++)
    {
        if (arr[i] > largest)
        {
            largest = arr[i];
        }
    }
    return largest;
}
---------------------------------------------------------------------------
4. Second Largest Element in Array
---------------------------------------------------------------------------
int SecondLargestArray(int arr[], int size)
{
    int largest = arr[0];
    int secondLargest = NULL;
    for (int i = 1; i < size; i++)
    {
        if (arr[i] > largest)
        {
            secondLargest = largest;
            largest = arr[i];
        }
        else if (arr[i] > secondLargest)
        {
            secondLargest = arr[i];
        }
    }
    return secondLargest;
}
---------------------------------------------------------------------------
5. Check if Array is sorted or not
---------------------------------------------------------------------------
bool isSorted(int arr[], int size)
{
    for (int i = 0; i < size - 1; i++)
    {
        if (arr[i] > arr[i + 1])
        {
            return false;
        }
    }
    return true;
}
---------------------------------------------------------------------------
6. Remove Duplicate from Sorted Array
---------------------------------------------------------------------------
void removeDupicate(int arr[], int size)
{
    int j = 0;
    for (int i = 1; i < size; i++)
    {
        if (arr[j] != arr[i])
        {
            arr[j + 1] = arr[i];
            j++;
        }
    }
    for (j; j < size; j++)
    {
        arr[j] = -1;
    }
}
